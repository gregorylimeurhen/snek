\section{Introduction}

World models have been proposed as an approach to human-level AI,
and multiplayer Snake was a community research request of OpenAI.
As our research contribution,
we investigate the use of world models for singleplayer Snake.

\section{Methodology}

\subsection{Data}

Let $\mathrm{L}\in\N_{\geq 2}$ be a grid extent,
and $\mathrm{D}=\mathrm{L}^{2}-1$ be a depth,
and $\mathcal{G}=[\mathrm{L}]^{2}$ be a grid,
and $\preceq_{\mathrm{row}}$ be the row-major order on $\mathcal{G}$,
and $\mathrm{U}\in\N_{\geq 1}$ be an upsampling factor,
and $\mathcal{X}=\R_{[0,1]}^{\mathrm{L}\mathrm{U}\times\mathrm{L}\mathrm{U}\times 3}$ be an image space.
% An upsampling factor allows us to talk about the game as an abstract grid purely using L because implementation can just take UL to scale up perfectly with rasterisation

\begin{definition}[Action Space]
$$
	\mathcal{A}=\{(0,0),(0,1),(0,-1),(-1,0),(1,0)\}
$$
\end{definition}

\begin{definition}[Snake Space]
$$
	\mathcal{S}
	=
	\bigcup_{n\geq 2}\{
		\bm{s}\in(\Z\times\Z)^{n}
		\mid
		\forall i\in[n-1](
			s_{i+1}-s_{i}
			\in
			\mathcal{A}
			\setminus
			\{(0,0)\}
		)
	\}
$$
\end{definition}

\begin{algorithm}[H]
\caption{$\textsc{Simulate}(\delta)$}
\begin{algorithmic}
	\Require $\delta:\N\to\mathcal{A}$
	\State ${
		\bm{s}
		\sim
		\mathcal{U}(
			\{
				\bm{s}
				:
				[2]
				\hookrightarrow
				\mathcal{G}
				\mid
				s_{2}-s_{1}
				\in
				\mathcal{A}
				\setminus
				\{(0,0)\}
			\}
		)
	}$
	\State ${
		E
		\gets
		\mathcal{G}
		\setminus
		\{
			s_{i}
			\mid
			i\in[\abs{\bm{s}}]
		\}
	}$
	\If{$E=\varnothing$}
		\State $f\gets\mathord{\perp}$
	\Else{}
		\State $f\gets\min_{\preceq_{\mathrm{row}}}E$
	\EndIf
	\State $t\gets 0$
	\While{$\bm{s}\neq\mathord{\perp}$}
		\State $i_{f}\gets 0$
		\State $d\gets\delta(t)$
		\If{$d=(0,0)$}
			\State $h\gets 2s_{1}-s_{2}$
		\Else{}
			\State $h\gets s_{1}+d$
		\EndIf
		\If{$h=f$}
			\State ${
				\bm{s}
				\gets
				h
				\parallel
				(s_{i})_{i=1}^{\abs{\bm{s}}}
			}$
			\State $i_{f}\gets 1$
		\Else{}
			\State ${
				\bm{s}
				\gets
				h
				\parallel
				(s_{i})_{i=1}^{\abs{\bm{s}}-1}
			}$
		\EndIf

		\State ${
			B
			\gets
			\{
				s_{i}
				\mid
				i\in[2..\abs{\bm{s}}]
			\}
		}$
		\If{$s_{1}\not\in\mathcal{G}$}
			\State $\bm{s}\gets\mathord{\perp}$
		\EndIf
		\If{$\bm{s}\neq\mathord{\perp}\land s_{1}\in B$}
			\State $\bm{s}\gets\mathord{\perp}$
		\EndIf

		\If{$\bm{s}\neq\mathord{\perp}\land i_{f}=1$}
			\State ${
				E
				\gets
				\mathcal{G}
				\setminus
				\mleft\{
					s_{i}
					\mid
					i\in[\abs{\bm{s}}]
				\mright\}
			}$
			\If{$E=\varnothing$}
				\State $f\gets\mathord{\perp}$
			\Else{}
				\State $f\gets\min_{\preceq_{\mathrm{row}}}E$
			\EndIf
		\EndIf

		\State $t\gets t+1$
	\EndWhile
\end{algorithmic}
\end{algorithm}

\subsection{Architecture}

Let $\mathrm{H}\in\N_{\geq 1}$ be a hidden dimensionality,
and $\mathcal{H}=\R^{\mathrm{H}}$ be a hidden space.

\begin{definition}[Encoder]
$$
	\phi:\mathcal{X}\to\mathcal{H}
$$
\end{definition}

\begin{definition}[World Model]
\begin{align*}
	\mathscr{F}
	&:
	\mathcal{H}\times\mathcal{A}
	\to
	\mathcal{H}
\end{align*}
\end{definition}

\begin{definition}[Evaluator]
\begin{align*}
	% "split into two heads, then merge into a sum with coefficient"
	\mathscr{V}_\mathrm{survival}
	&:
	\mathcal{H}^{\mathrm D} % trajectory
	\to
	\R_{[0,1]} % probability of trajectory survival
	,\ 
	\mathscr{V}_\mathrm{consumption}
	:
	\mathcal{H}^{\mathrm D} % trajectory
	\to
	\R_{[0,1]}^{\mathrm D} % probability of consumption at each state of trajectory
	\\
	\mathscr{V}
	&:
	\mathcal{H}^{\mathrm D} % trajectory
	\to
	\R_{\geq 0}
	,\ 
	\mathscr{V}(\bm{h})
	=
	\mathscr{V}_\mathrm{survival}(\bm{h})
	\bm{\sum}
	\mathscr{V}_\mathrm{consumption}(\bm{h})
\end{align*}
\end{definition}

% The setup, from end to end, must be:
% Encoder -> World Model -> Evaluator
% where V_survival is trained to approximate trajectory survival binary
% and V_consumption is trained to approximate consumption binaries

To avoid representation collapse, we apply SIGReg to encoder outputs by batching
$$
	\mathcal{L}_\mathrm{encoder}
	=
	\frac{1}{2}
	\operatorname{SIGReg}(\{\phi(x_{t})\mid t\in B\})
	+
	\frac{1}{2}
	\operatorname{SIGReg}(\{\phi(x_{t+1})\mid t\in B\})
$$
to encourage representations to follow an approximately isotropic Gaussian distribution while preserving their usefulness for transition prediction.

% SIGReg is a fairly new technique: https://arxiv.org/pdf/2511.08544)
% In essence, we have:
% \lambda_\mathrm{encoder}\mathcal{L}_\mathrm{encoder}
% + \lambda_\mathrm{wm}\mathcal{L}_\mathrm{wm}
% + \lambda_\mathrm{evaluator}\mathcal{L}_\mathrm{evaluator}

We use AdamW optimiser.
\begin{itemize}
	\item Encoder: tiny CNN to latent state.
	\item World model: residual action-conditioned MLP in latent space.
	\item Evaluator: GRU with survival and consumption heads.
	\item Loss: latent MSE for the world model, binary cross-entropy for evaluator heads, and light SIGReg on encoder outputs.
\end{itemize}

At inference, given an image $x\in\mathcal{X}$, we find
$$
	\arg\max_{\bm{a}\in\operatorname{legal}(\mathcal{A})^{\mathrm{D}}}
	\mathscr{V}\mleft(
		[
			f_{1}=\mathscr{F}(\phi(x), a_{1}),
			f_{2}=\mathscr{F}(f_{1}, a_{2}),
			\dots,
			f_{\mathrm{D}}=\mathscr{F}(f_{\mathrm{D}-1},a_{\mathrm{D}})
		]
	\mright)
$$
where an action is "legal" if it can be made in a game;
for instance, if a snake (head) is at a cell at the top edge of the grid,
then "up" is legal (albeit bad/losing), and "up, up", "up, right", etc. are \textbf{illegal}.
% The planner must not use illegal actions which cannot possibly have been executed during training

\begin{notation}
Let $n\in\N_{\geq 1}$.
$$
	[n]=\N_{[1,n]}
$$
\end{notation}

\begin{notation}
Let $a,b\in\N_{\geq 1}$.
$$
	[a..b]=\N_{[a,b]}
$$
\end{notation}

\begin{notation}
Let $x$ be a set, and $\bm{y}$ be a sequence.
$$
	x\bm{\in}\bm{y}
	\equiv
	x\in\im(\bm{y})
$$
\end{notation}

\begin{notation}
Let $\bm{x},\bm{y}$ be sequences.
$$
	\bm{x}\bm{\subseteq}\bm{y}
	\equiv
	\im(\bm{x})\subseteq\im(\bm{y})
$$
\end{notation}

\begin{notation}
Let $\bm{x}$ be a sequence, and $y$ be a set.
$$
	\bm{x}\bm{\setminus}y
	=
	(x_{i})_{i\in\{j\in\dom(\bm{x})\mid x_{j}\not\in y\}}
$$
\end{notation}

\begin{notation}
Let $\bm{x}$ be a sequence.
$$
	\abs{\bm{x}}
	=
	\abs{\dom(\bm{x})}
$$
\end{notation}

\begin{notation}
Let $n\geq 1$, and $X$ be a set, and $f_{1},\dots,f_{n}:X\to X$.
$$
	\bigcirc_{i=1}^{n}f_{i}
	=
	f_{n}\circ\dots\circ f_{1}
$$
\end{notation}
